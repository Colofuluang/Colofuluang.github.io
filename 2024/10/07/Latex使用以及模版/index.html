Latex 使用以及模版

2024/10/7 Latex

这里存放的是我的 overleaf 模版。

1. 设置编译器

设置为 XeLatex 主题，导入图片。

2. 模版

模版内容如下。

\documentclass{article}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% \tiny：最小的字体大小。
% \scriptsize：比 \tiny 大一点，用于显示脚注或者表格中的文字。
% \footnotesize：比 \scriptsize 大一点，用于显示正文中的脚注。
% \small：比 \footnotesize 大一点，用于显示正文中较小的文字。
% \normalsize：默认的正文字体大小。
% \large：比 \normalsize 大一点。
% \Large：比 \large 大一点。
% \LARGE：比 \Large 大一点。
% \huge：比 \LARGE 大一点。
% \Huge：最大的字体大小。
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% package
\usepackage{graphicx} % Required for inserting images
\usepackage{xeCJK}
\usepackage[hidelinks]{hyperref}
\usepackage{caption}
\usepackage{setspace} % section 缩进
\usepackage{changepage} % 局部缩进
\setstretch{1.5} % 调整正文行距
\usepackage{amsmath} % 数学公式
\usepackage{textcomp} % 圆圈数字①
\usepackage{multirow} % 表格划线
\usepackage{svg} % 插入svg图像
\usepackage{pdfpages} % 插入pdf
\usepackage[T1]{fontenc} % 使用T1字体编码
\usepackage[utf8]{inputenc} % 使用UTF-8编码
\usepackage{enumitem}
\usepackage{csquotes} % csquotes宏包，用于处理引号
\usepackage{algorithm} % 算法宏包，伪代码
\usepackage{algorithmic}
% 设置西文双引号和单引号的字体
\renewcommand{\mkblockquote}[4]{
    \textquotedblleft#1\textquotedblright#2\textquoteleft#3\textquoteright#4
}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% 设置页面布局，调整边距
\usepackage{geometry}
\geometry{
    a4paper,
    left=2cm,
    right=2cm,
    top=3cm,
    bottom=3cm,
}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% 设置代码框样式
\usepackage{listings}
\usepackage{xcolor}

\definecolor{lightgray}{gray}{0.95}
% 设置代码框的样式
% BASH
\lstdefinestyle{bash}{
    backgroundcolor=\color{lightgray},
    basicstyle=\ttfamily,
    breaklines=true,
    captionpos=b,
    frame=single,
    numbers=left,
    numbersep=5pt,
    numberstyle=\tiny\color{gray},
    showstringspaces=false,
    tabsize=2
}
% Python
\lstdefinestyle{pycharm}{
    backgroundcolor=\color{lightgray},
    basicstyle=\ttfamily,
    breaklines=true,
    captionpos=b,
    basicstyle=\footnotesize\ttfamily\color{black},
    rulecolor=\color{black},
    numbers=left,
    numbersep=5pt,
    numberstyle=\tiny\color{gray},
    showstringspaces=false,
    tabsize=2
}
% 定义 C 语言风格
\lstdefinestyle{customc}{
  backgroundcolor=\color{lightgray},
  belowcaptionskip=1\baselineskip,
  breaklines=true,
  frame=L,
  xleftmargin=\parindent,
  language=C,
  showstringspaces=false,
  basicstyle=\footnotesize\ttfamily\color{black},
  keywordstyle=\bfseries\color{green!40!black},
  commentstyle=\itshape\color{purple!40!black},
  identifierstyle=\color{blue},
  stringstyle=\color{orange},
}
% 定义 IDEA 样式
\lstdefinestyle{IDEA}{
    backgroundcolor=\color{white},
    basicstyle=\small\ttfamily,
    breakatwhitespace=false,
    breaklines=true,
    captionpos=b,
    commentstyle=\color{green!40!black},
    frame=single,
    keywordstyle=\color{blue},
    language=Java,
    numbers=left,
    numbersep=5pt,
    numberstyle=\tiny\color{gray},
    rulecolor=\color{black},
    showspaces=false,
    showstringspaces=false,
    showtabs=false,
    stringstyle=\color{red},
    tabsize=4
}
\captionsetup[lstlisting]{position=bottom}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% 目录首行
\renewcommand{\contentsname}{目录}
% section编号
\usepackage{zhnumber} % change section number to chinese
\renewcommand\thesection{\zhnum{section}}
\renewcommand\thesubsection{\hspace{1em}\arabic{subsection}.}
\renewcommand\thesubsubsection{\hspace{2em}\Roman{subsubsection}}
\newcommand{\Uproman}[1]{\MakeUppercase{\romannumeral#1}}
% 设置目录中显示到第2级section
\setcounter{tocdepth}{2} % 目录
% \setcounter{secnumdepth}{2} % 正文
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% title
\title{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% *可在前面无空行时，避免自动排版
\vspace*{2\baselineskip}
\begin{figure}[ht]
    \centering
    \includegraphics[width=0.7\textwidth]{logo.png}
\end{figure}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\vspace*{2\baselineskip}
\huge
课程名称
\Large

\vspace*{1\baselineskip}
实验/作业项目题目
\vspace*{1\baselineskip}
\begin{table}[!ht]
    \centering
    \large
    \begin{tabular}{lll}
        \hline
        \multicolumn{1}{c}{姓名} & \multicolumn{1}{c}{学号} &  \\ \hline
        高洋洋 & 21312049 \\ \hline   
    \end{tabular}
\end{table}
}
\date{日期}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
